# @luzhongk/vue-dialog

一个简单 vue 弹窗

# 安装

yarn add @luzhongk/vue-dialog

# 使用

import VueDialog from '@luzhongk/vue-dialog'

export default {
  components: {
    VueDialog
  },
  data() {
    return {
      visible: false
    }
  }
}

<div>
  <vue-dialog v-model="visible" />
</div>
